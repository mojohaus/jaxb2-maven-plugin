203     @Test
204     public void validateProcessingNodes()
205     {
206 
207         // Assemble
208         final String newPrefix = "changedFoo";
209         final String oldPrefix = "foo";
210         final String originalXml = getXmlDocumentSample( oldPrefix );
211         final String changedXml = getXmlDocumentSample( newPrefix );
212         final NodeProcessor changeNamespacePrefixProcessor = new ChangeNamespacePrefixProcessor( oldPrefix, newPrefix );
213 
214         // Act
215         final Document processedDocument = XsdGeneratorHelper.parseXmlStream( new StringReader( originalXml ) );
216         XsdGeneratorHelper.process( processedDocument.getFirstChild(), true, changeNamespacePrefixProcessor );
217 
218         // Assert

220         final Diff diff = new Diff( expectedDocument, processedDocument, null, new ElementNameAndAttributeQualifier() );
221         diff.overrideElementQualifier( new ElementNameAndAttributeQualifier() );
222 
223         XMLAssert.assertXMLEqual( processedDocument, expectedDocument );
224     }

226     @Test
227     public void validateProcessingXSDsWithEnumerations() throws Exception
228     {
229 
230         // Assemble
231         final BufferingLog log = new BufferingLog();
232         final JavaDocExtractor extractor = new JavaDocExtractor( log );
233         extractor.setEncoding( "UTF-8" );
234 
235         final String parentPath = "testdata/schemageneration/javadoc/enums/";
236         final URL parentPathURL = getClass().getClassLoader().getResource( parentPath );
237         Assert.assertNotNull( parentPathURL );
238 
239         final File parentDir = new File( parentPathURL.getPath() );
240         Assert.assertTrue( parentDir.exists() && parentDir.isDirectory() );
241 
242         final List<Filter<File>> excludeFilesMatching = new ArrayList<Filter<File>>();
243         excludeFilesMatching.add( new PatternFileFilter( Collections.singletonList( "\\.xsd" ) ) );
244         Filters.initialize( log, excludeFilesMatching );
245 
246         final List<File> allSourceFiles = FileSystemUtilities.filterFiles( parentDir, null, parentDir.getAbsolutePath(),
247                 log, "allJavaFiles", excludeFilesMatching );
248         Assert.assertEquals( 3, allSourceFiles.size() );
249 
250         final List<URL> urls = new ArrayList<URL>();
251         for ( File current : allSourceFiles )
252         {
253             try
254             {
255                 urls.add( current.toURI().toURL() );
256             }
257             catch ( MalformedURLException e )
258             {
259                 throw new IllegalArgumentException(
260                         "Could not convert file [" + current.getAbsolutePath() + "] to a URL", e );
261             }
262         }
263         Assert.assertEquals( 3, urls.size() );
264 
265         extractor.addSourceURLs( urls );
266         final SearchableDocumentation docs = extractor.process();
267 
268         // #1) The raw / un-processed XSD (containing the 'before' state)
269         // #2) The processed XSD (containing the 'expected' state)
270         final String rawEnumSchema = PropertyResources.readFully( parentPath + "rawEnumSchema.xsd" );
271         final String processedEnumSchema = PropertyResources.readFully( parentPath + "processedEnumSchema.xsd" );
272         final NodeProcessor enumProcessor = new XsdEnumerationAnnotationProcessor( docs,
273                 new NoAuthorJavaDocRenderer() );
274 
275         // Act
276         final Document processedDocument = XsdGeneratorHelper.parseXmlStream( new StringReader( rawEnumSchema ) );
277         XsdGeneratorHelper.process( processedDocument.getFirstChild(), true, enumProcessor );
278         // System.out.println("Got: " + AbstractSourceCodeAwareNodeProcessingTest.printDocument(processedDocument));
279 
280         // Assert
281         final Document expectedDocument = XsdGeneratorHelper.parseXmlStream( new StringReader( processedEnumSchema ) );
282         final Diff diff = new Diff( expectedDocument, processedDocument, null, new ElementNameAndAttributeQualifier() );
283         diff.overrideElementQualifier( new ElementNameAndAttributeQualifier() );
284 
285         XMLAssert.assertXMLEqual( processedDocument, expectedDocument );
286     }

343     @Test
344     public void validateAcquiringFilenameToResolverMap() throws MojoExecutionException
345     {
346 
347         // Assemble
348         final String[] expectedFilenames = {"schema1.xsd", "schema2.xsd", "schema3.xsd"};
349         final URL tmpUrl = getClass().getClassLoader().getResource( "generated/schema/schema1.xsd" );
350         final File directory = new File( tmpUrl.getFile() ).getParentFile();
351 
352         // Act
353         final Map<String, SimpleNamespaceResolver> fileNameToResolverMap = XsdGeneratorHelper.getFileNameToResolverMap(
354                 directory );
355 
356         // Assert
357         Assert.assertEquals( 3, fileNameToResolverMap.size() );
358         for ( String current : expectedFilenames )
359         {
360             Assert.assertTrue( fileNameToResolverMap.keySet().contains( current ) );
361         }
362 
363         SimpleNamespaceResolver schema1Resolver = fileNameToResolverMap.get( "schema1.xsd" );
364         Assert.assertEquals( "http://yet/another/namespace", schema1Resolver.getLocalNamespaceURI() );
365         Assert.assertEquals( "schema1.xsd", schema1Resolver.getSourceFilename() );
366         final Map<String, String> schema1NamespaceURI2PrefixMap = schema1Resolver.getNamespaceURI2PrefixMap();
367         Assert.assertEquals( 1, schema1NamespaceURI2PrefixMap.size() );
368         Assert.assertEquals( "xs", schema1NamespaceURI2PrefixMap.get( "http://www.w3.org/2001/XMLSchema" ) );
369 
370         SimpleNamespaceResolver schema2Resolver = fileNameToResolverMap.get( "schema2.xsd" );
371         Assert.assertEquals( "http://some/namespace", schema2Resolver.getLocalNamespaceURI() );
372         Assert.assertEquals( "schema2.xsd", schema2Resolver.getSourceFilename() );
373         final Map<String, String> schema2NamespaceURI2PrefixMap = schema2Resolver.getNamespaceURI2PrefixMap();
374         Assert.assertEquals( 2, schema2NamespaceURI2PrefixMap.size() );
375         Assert.assertEquals( "ns1", schema2NamespaceURI2PrefixMap.get( "http://another/namespace" ) );
376         Assert.assertEquals( "xs", schema2NamespaceURI2PrefixMap.get( "http://www.w3.org/2001/XMLSchema" ) );
377 
378         SimpleNamespaceResolver schema3Resolver = fileNameToResolverMap.get( "schema3.xsd" );
379         Assert.assertEquals( "http://another/namespace", schema3Resolver.getLocalNamespaceURI() );
380         Assert.assertEquals( "schema3.xsd", schema3Resolver.getSourceFilename() );
381         final Map<String, String> schema3NamespaceURI2PrefixMap = schema3Resolver.getNamespaceURI2PrefixMap();
382         Assert.assertEquals( 3, schema3NamespaceURI2PrefixMap.size() );
383         Assert.assertEquals( "ns2", schema3NamespaceURI2PrefixMap.get( "http://yet/another/namespace" ) );
384         Assert.assertEquals( "ns1", schema3NamespaceURI2PrefixMap.get( "http://some/namespace" ) );
385         Assert.assertEquals( "xs", schema3NamespaceURI2PrefixMap.get( "http://www.w3.org/2001/XMLSchema" ) );
386     }
