122     /**
123      * <p>Retrieves the canonical File matching the supplied path in the following order or priority:</p>
124      * <ol>
125      * <li><strong>Absolute path:</strong> The path is used by itself (i.e. {@code new File(path);}). If an
126      * existing file or directory matches the provided path argument, its canonical path will be returned.</li>
127      * <li><strong>Relative path:</strong> The path is appended to the baseDir (i.e.
128      * {@code new File(baseDir, path);}). If an existing file or directory matches the provided path argument,
129      * its canonical path will be returned. Only in this case will be baseDir argument be considered.</li>
130      * </ol>
131      * <p>If no file or directory could be derived from the supplied path and baseDir, {@code null} is returned.</p>
132      *
133      * @param path    A non-null path which will be used to find an existing file or directory.
134      * @param baseDir A directory to which the path will be appended to search for the existing file or directory in
135      *                case the file was nonexistent when interpreted as an absolute path.
136      * @return either a canonical File for the path, or {@code null} if no file or directory matched
137      * the supplied path and baseDir.
138      */
139     public static File getExistingFile(final String path, final File baseDir) {
140 
141         // Check sanity
142         Validate.notEmpty(path, "path");
143         final File theFile = new File(path);
144         File toReturn = null;
145 
146         // Is 'path' absolute?
147         if (theFile.isAbsolute() && (EXISTING_FILE.accept(theFile) || EXISTING_DIRECTORY.accept(theFile))) {
148             toReturn = getCanonicalFile(theFile);
149         }
150 
151         // Is 'path' relative?
152         if (!theFile.isAbsolute()) {
153 
154             // In this case, baseDir cannot be null.
155             Validate.notNull(baseDir, "baseDir");
156             final File relativeFile = new File(baseDir, path);
157 
158             if (EXISTING_FILE.accept(relativeFile) || EXISTING_DIRECTORY.accept(relativeFile)) {
159                 toReturn = getCanonicalFile(relativeFile);
160             }
161         }
162 
163         // The path provided did not point to an existing File or Directory.
164         return toReturn;
165     }

168      * Retrieves the URL for the supplied File. Convenience method which hides exception handling
169      * for the operation in question.
170      *
171      * @param aFile A File for which the URL should be retrieved.
172      * @return The URL for the supplied aFile.
173      * @throws java.lang.IllegalArgumentException if getting the URL yielded a MalformedURLException.
174      */
175     public static URL getUrlFor(final File aFile) throws IllegalArgumentException {
176 
177         // Check sanity
178         Validate.notNull(aFile, "aFile");
179 
180         try {
181             return aFile.toURI().normalize().toURL();
182         } catch (MalformedURLException e) {
183             throw new IllegalArgumentException("Could not retrieve the URL from file ["
184                     + getCanonicalPath(aFile) + "]", e);
185         }

188     /**
189      * Acquires the file for a supplied URL, provided that its protocol is is either a file or a jar.
190      *
191      * @param anURL    a non-null URL.
192      * @param encoding The encoding to be used by the URLDecoder to decode the path found.
193      * @return The File pointing to the supplied URL, for file or jar protocol URLs and null otherwise.
194      */
195     public static File getFileFor(final URL anURL, final String encoding) {
196 
197         // Check sanity
198         Validate.notNull(anURL, "anURL");
199         Validate.notNull(encoding, "encoding");
200 
201         final String protocol = anURL.getProtocol();
202         File toReturn = null;
203         if ("file".equalsIgnoreCase(protocol)) {
204             try {
205                 final String decodedPath = URLDecoder.decode(anURL.getPath(), encoding);
206                 toReturn = new File(decodedPath);
207             } catch (Exception e) {
208                 throw new IllegalArgumentException("Could not get the File for [" + anURL + "]", e);
209             }
210         } else if ("jar".equalsIgnoreCase(protocol)) {
211 
212             try {
213 
214                 // Decode the JAR
215                 final String tmp = URLDecoder.decode(anURL.getFile(), encoding);
216 
217                 // JAR URLs generally contain layered protocols, such as:
218                 // jar:file:/some/path/to/nazgul-tools-validation-aspect-4.0.2.jar!/the/package/ValidationAspect.class
219                 final URL innerURL = new URL(tmp);
220 
221                 // We can handle File protocol URLs here.
222                 if ("file".equalsIgnoreCase(innerURL.getProtocol())) {
223 
224                     // Peel off the inner protocol
225                     final String innerUrlPath = innerURL.getPath();
226                     final String filePath = innerUrlPath.contains("!")
227                             ? innerUrlPath.substring(0, innerUrlPath.indexOf("!"))
228                             : innerUrlPath;
229                     toReturn = new File(URLDecoder.decode(filePath, encoding));
230                 }
231             } catch (Exception e) {
232                 throw new IllegalArgumentException("Could not get the File for [" + anURL + "]", e);
233             }
234         }
235 
236         // All done.
237         return toReturn;
238     }

241     /**
242      * Filters files found either in the sources paths (or in the standardDirectory if no explicit sources are given),
243      * and retrieves a List holding those files that do not match any of the supplied Java Regular Expression
244      * excludePatterns.
245      *
246      * @param baseDir             The non-null basedir Directory.
247      * @param sources             The sources which should be either absolute or relative (to the given baseDir)
248      *                            paths to files or to directories that should be searched recursively for files.
249      * @param standardDirectories If no sources are given, revert to searching all files under these standard
250      *                            directories. Each path is {@code relativize()}-d to the supplied baseDir to
251      *                            reach a directory path.
252      * @param log                 A non-null Maven Log for logging any operations performed.
253      * @param fileTypeDescription A human-readable short description of what kind of files are searched for, such as
254      *                            "xsdSources" or "xjbSources".
255      * @param excludePatterns     An optional List of patterns used to construct an ExclusionRegExpFileFilter used to
256      *                            identify files which should be excluded from the result.
257      * @return URLs to all Files under the supplied sources (or standardDirectories, if no explicit sources
258      * are given) which do not match the supplied Java Regular excludePatterns.
259      */
260     @SuppressWarnings("all")
261     public static List<URL> filterFiles(final File baseDir,
262                                         final List<String> sources,
263                                         final List<String> standardDirectories,
264                                         final Log log,
265                                         final String fileTypeDescription,
266                                         final List<Filter<File>> excludePatterns) {
267 
268         final SortedMap<String, File> pathToResolvedSourceMap = new TreeMap<String, File>();
269 
270         for (String current : standardDirectories) {
271             for (File currentResolvedSource : FileSystemUtilities.filterFiles(
272                     baseDir,
273                     sources,
274                     FileSystemUtilities.relativize(current, baseDir, true),
275                     log,
276                     fileTypeDescription,
277                     excludePatterns)) {
278 
279                 // Add the source
280                 pathToResolvedSourceMap.put(
281                         FileSystemUtilities.getCanonicalPath(currentResolvedSource),
282                         currentResolvedSource);
283             }
284         }
285 
286         final List<URL> toReturn = new ArrayList<URL>();
287 
288         // Extract the URLs for all resolved Java sources.
289         for (Map.Entry<String, File> current : pathToResolvedSourceMap.entrySet()) {
290             toReturn.add(FileSystemUtilities.getUrlFor(current.getValue()));
291         }
292 
293         if (log.isDebugEnabled()) {
294 
295             final StringBuilder builder = new StringBuilder();
296             builder.append("\n+=================== [Filtered " + fileTypeDescription + "]\n");
297 
298             builder.append("|\n");
299             builder.append("| " + excludePatterns.size() + " Exclude patterns:\n");
300             for (int i = 0; i < excludePatterns.size(); i++) {
301                 builder.append("| [" + (i + 1) + "/" + excludePatterns.size() + "]: " + excludePatterns.get(i) + "\n");
302             }
303 
304             builder.append("|\n");
305             builder.append("| " + standardDirectories.size() + " Standard Directories:\n");
306             for (int i = 0; i < standardDirectories.size(); i++) {
307                 builder.append("| [" + (i + 1) + "/" + standardDirectories.size() + "]: "
308                         + standardDirectories.get(i) + "\n");
309             }
310 
311             builder.append("|\n");
312             builder.append("| " + toReturn.size() + " Results:\n");
313             for (int i = 0; i < toReturn.size(); i++) {
314                 builder.append("| [" + (i + 1) + "/" + toReturn.size() + "]: " + toReturn.get(i) + "\n");
315             }
316             builder.append("|\n");
317             builder.append("+=================== [End Filtered " + fileTypeDescription + "]\n\n");
318 
319             // Log all.
320             log.debug(builder.toString().replace("\n", AbstractJaxbMojo.NEWLINE));
321         }
322 
323         // All done.
324         return toReturn;
325     }

327     /**
328      * Filters files found either in the sources paths (or in the standardDirectory if no explicit sources are given),
329      * and retrieves a List holding those files that do not match any of the supplied Java Regular Expression
330      * excludePatterns.
331      *
332      * @param baseDir             The non-null basedir Directory.
333      * @param sources             The sources which should be either absolute or relative (to the given baseDir)
334      *                            paths to files or to directories that should be searched recursively for files.
335      * @param standardDirectory   If no sources are given, revert to searching all files under this standard directory.
336      *                            This is the path appended to the baseDir to reach a directory.
337      * @param log                 A non-null Maven Log for logging any operations performed.
338      * @param fileTypeDescription A human-readable short description of what kind of files are searched for, such as
339      *                            "xsdSources" or "xjbSources".
340      * @param excludeFilters      An optional List of Filters used to identify files which should be excluded from
341      *                            the result.
342      * @return All files under the supplied sources (or standardDirectory, if no explicit sources are given) which
343      * do not match the supplied Java Regular excludePatterns.
344      */
345     @SuppressWarnings("CheckStyle")
346     public static List<File> filterFiles(final File baseDir,
347                                          final List<String> sources,
348                                          final String standardDirectory,
349                                          final Log log,
350                                          final String fileTypeDescription,
351                                          final List<Filter<File>> excludeFilters) {
352 
353         // Check sanity
354         Validate.notNull(baseDir, "baseDir");
355         Validate.notNull(log, "log");
356         Validate.notEmpty(standardDirectory, "standardDirectory");
357         Validate.notEmpty(fileTypeDescription, "fileTypeDescription");
358 
359         // No sources provided? Fallback to the standard (which should be a relative path).
360         List<String> effectiveSources = sources;
361         if (sources == null || sources.isEmpty()) {
362             effectiveSources = new ArrayList<String>();
363 
364             final File tmp = new File(standardDirectory);
365             final File rootDirectory = tmp.isAbsolute() ? tmp : new File(baseDir, standardDirectory);
366             effectiveSources.add(FileSystemUtilities.getCanonicalPath(rootDirectory));
367         }
368 
369         // First, remove the non-existent sources.
370         List<File> existingSources = new ArrayList<File>();
371         for (String current : effectiveSources) {
372 
373             final File existingFile = FileSystemUtilities.getExistingFile(current, baseDir);
374             if (existingFile != null) {
375                 existingSources.add(existingFile);
376 
377                 if (log.isDebugEnabled()) {
378                     log.debug("Accepted configured " + fileTypeDescription + " ["
379                             + FileSystemUtilities.getCanonicalFile(existingFile) + "]");
380                 }
381             } else {
382                 if (log.isInfoEnabled()) {
383                     log.info("Ignored given or default " + fileTypeDescription + " [" + current
384                             + "], since it is not an existent file or directory.");
385                 }
386             }
387         }
388 
389         if (log.isDebugEnabled() && existingSources.size() > 0) {
390 
391             final int size = existingSources.size();
392 
393             log.debug(" [" + size + " existing " + fileTypeDescription + "] ...");
394             for (int i = 0; i < size; i++) {
395                 log.debug("   " + (i + 1) + "/" + size + ": " + existingSources.get(i));
396             }
397             log.debug(" ... End [" + size + " existing " + fileTypeDescription + "]");
398         }
399 
400         // All Done.
401         return FileSystemUtilities.resolveRecursively(existingSources, excludeFilters, log);
402     }

404     /**
405      * Filters all supplied files using the provided {@code acceptFilter}.
406      *
407      * @param files        The list of files to resolve, filter and return. If the {@code files} List
408      *                     contains directories, they are searched for Files recursively. Any found Files in such
409      *                     a search are included in the resulting File List if they match the acceptFilter supplied.
410      * @param acceptFilter A filter matched to all files in the given List. If the acceptFilter matches a file, it is
411      *                     included in the result.
412      * @param log          The active Maven Log.
413      * @return All files in (or files in subdirectories of directories provided in) the files List, provided that each
414      * file is accepted by an ExclusionRegExpFileFilter.
415      */
416     public static List<File> filterFiles(final List<File> files, final Filter<File> acceptFilter, final Log log) {
417 
418         // Check sanity
419         Validate.notNull(files, "files");
420 
421         final List<File> toReturn = new ArrayList<File>();
422 
423         if (files.size() > 0) {
424             for (File current : files) {
425 
426                 final boolean isAcceptedFile = EXISTING_FILE.accept(current) && acceptFilter.accept(current);
427                 final boolean isAcceptedDirectory = EXISTING_DIRECTORY.accept(current) && acceptFilter.accept(current);
428 
429                 if (isAcceptedFile) {
430                     toReturn.add(current);
431                 } else if (isAcceptedDirectory) {
432                     recurseAndPopulate(toReturn, Collections.singletonList(acceptFilter), current, false, log);
433                 }
434             }
435         }
436 
437         // All done
438         return toReturn;
439     }

441     /**
442      * Retrieves a List of Files containing all the existing files within the supplied files List, including all
443      * files found in directories recursive to any directories provided in the files list. Each file included in the
444      * result must pass an ExclusionRegExpFileFilter synthesized from the supplied exclusions pattern(s).
445      *
446      * @param files            The list of files to resolve, filter and return. If the {@code files} List
447      *                         contains directories, they are searched for Files recursively. Any found Files in such
448      *                         a search are included in the resulting File List if they do not match any of the
449      *                         exclusionFilters supplied.
450      * @param exclusionFilters A List of Filters which identify files to remove from the result - implying that any
451      *                         File matched by any of these exclusionFilters will not be included in the result.
452      * @param log              The active Maven Log.
453      * @return All files in (or files in subdirectories of directories provided in) the files List, provided that each
454      * file is accepted by an ExclusionRegExpFileFilter.
455      */
456     public static List<File> resolveRecursively(final List<File> files,
457                                                 final List<Filter<File>> exclusionFilters,
458                                                 final Log log) {
459 
460         // Check sanity
461         Validate.notNull(files, "files");
462 
463         final List<Filter<File>> effectiveExclusions = exclusionFilters == null
464                 ? new ArrayList<Filter<File>>()
465                 : exclusionFilters;
466 
467         final List<File> toReturn = new ArrayList<File>();
468 
469         if (files.size() > 0) {
470             for (File current : files) {
471 
472                 final boolean isAcceptedFile = EXISTING_FILE.accept(current)
473                         && Filters.noFilterMatches(current, effectiveExclusions);
474                 final boolean isAcceptedDirectory = EXISTING_DIRECTORY.accept(current)
475                         && Filters.noFilterMatches(current, effectiveExclusions);
476 
477                 if (isAcceptedFile) {
478                     toReturn.add(current);
479                 } else if (isAcceptedDirectory) {
480                     recurseAndPopulate(toReturn, effectiveExclusions, current, true, log);
481                 }
482             }
483         }
484 
485         // All done
486         return toReturn;
487     }

489     /**
490      * Convenience method to successfully create a directory - or throw an exception if failing to create it.
491      *
492      * @param aDirectory        The directory to create.
493      * @param cleanBeforeCreate if {@code true}, the directory and all its content will be deleted before being
494      *                          re-created. This will ensure that the created directory is really clean.
495      * @throws MojoExecutionException if the aDirectory could not be created (and/or cleaned).
496      */
497     public static void createDirectory(final File aDirectory, final boolean cleanBeforeCreate)
498             throws MojoExecutionException {
499 
500         // Check sanity
501         Validate.notNull(aDirectory, "aDirectory");
502         validateFileOrDirectoryName(aDirectory);
503 
504         // Clean an existing directory?
505         if (cleanBeforeCreate) {
506             try {
507                 FileUtils.deleteDirectory(aDirectory);
508             } catch (IOException e) {
509                 throw new MojoExecutionException("Could not clean directory [" + getCanonicalPath(aDirectory) + "]", e);
510             }
511         }
512 
513         // Now, make the required directory, if it does not already exist as a directory.
514         final boolean existsAsFile = aDirectory.exists() && aDirectory.isFile();
515         if (existsAsFile) {
516             throw new MojoExecutionException("[" + getCanonicalPath(aDirectory) + "] exists and is a file. "
517                     + "Cannot make directory");
518         } else if (!aDirectory.exists() && !aDirectory.mkdirs()) {
519             throw new MojoExecutionException("Could not create directory [" + getCanonicalPath(aDirectory) + "]");
520         }
521     }

523     /**
524      * If the supplied path refers to a file or directory below the supplied basedir, the returned
525      * path is identical to the part below the basedir.
526      *
527      * @param path      The path to strip off basedir path from, and return.
528      * @param parentDir The maven project basedir.
529      * @param removeInitialFileSep If true, an initial {@code File#separator} is removed before returning.
530      * @return The path relative to basedir, if it is situated below the basedir. Otherwise the supplied path.
531      */
532     public static String relativize(final String path,
533                                     final File parentDir,
534                                     final boolean removeInitialFileSep) {
535 
536         // Check sanity
537         Validate.notNull(path, "path");
538         Validate.notNull(parentDir, "parentDir");
539 
540         final Path p = Paths.get(path);
541         final Path pd = parentDir.toPath();
542 
543         String platformSpecificPath;
544         if (p.normalize().startsWith(pd.normalize().toString())) {
545             platformSpecificPath = pd.relativize(p).toString();
546         } else {
547             platformSpecificPath = p.toString();
548         }
549 
550         if (removeInitialFileSep && platformSpecificPath.startsWith(File.separator)) {
551             platformSpecificPath = platformSpecificPath.substring(File.separator.length());
552         }
553 
554         // NOTE: it appears this function is meant to preserve the file separator that was passed in the path
555         if (path.indexOf('\\') == -1) {
556             platformSpecificPath = platformSpecificPath.replace('\\', '/');
557         }
558 
559         return platformSpecificPath;
560     }

581     /**
582      * If the supplied fileOrDir is a File, it is added to the returned List if any of the filters Match.
583      * If the supplied fileOrDir is a Directory, it is listed and any of the files immediately within the fileOrDir
584      * directory are returned within the resulting List provided that they match any of the supplied filters.
585      *
586      * @param fileOrDir              A File or Directory.
587      * @param fileFilters            A List of filter of which at least one must match to add the File (or child Files, in case
588      *                               of a directory) to the resulting List.
589      * @param excludeFilterOperation if {@code true}, all fileFilters are considered exclude filter, i.e. if
590      *                               any of the Filters match the fileOrDir, that fileOrDir will be excluded from the
591      *                               result.
592      * @param log                    The active Maven Log
593      * @return A List holding the supplied File (or child Files, in case fileOrDir is a Directory) given that at
594      * least one Filter accepts them.
595      */
596     @SuppressWarnings("all")
597     public static List<File> listFiles(final File fileOrDir,
598                                        final List<Filter<File>> fileFilters,
599                                        final boolean excludeFilterOperation,
600                                        final Log log) {
601 
602         // Check sanity
603         Validate.notNull(log, "log");
604         Validate.notNull(fileFilters, "fileFilters");
605         final List<File> toReturn = new ArrayList<File>();
606 
607         if (EXISTING_FILE.accept(fileOrDir)) {
608             checkAndAdd(toReturn, fileOrDir, fileFilters, excludeFilterOperation, log);
609         } else if (EXISTING_DIRECTORY.accept(fileOrDir)) {
610 
611             final File[] listedFiles = fileOrDir.listFiles();
612 
613             if (listedFiles != null) {
614                 for (File current : listedFiles) {
615 
616                     // Typically, hidden directories start with '.'
617                     // Except them from automagic filtering.
618                     if (current.getName().charAt(0) != '.') {
619                         checkAndAdd(toReturn, current, fileFilters, excludeFilterOperation, log);
620                     }
621                 }
622             }
623         }
624 
625         // All done.
626         return toReturn;
627     }

629     //
630     // Private helpers
631     //
632 
633     private static void checkAndAdd(final List<File> toPopulate,
634                                     final File current,
635                                     final List<Filter<File>> fileFilters,
636                                     final boolean excludeFilterOperation,
637                                     final Log log) {
638 
639         //
640         // When no filters are supplied...
641         // [Include Operation]: all files will be rejected
642         // [Exclude Operation]: all files will be included
643         //
644         final boolean noFilters = fileFilters == null || fileFilters.isEmpty();
645         final boolean addFile = excludeFilterOperation
646                 ? noFilters || Filters.rejectAtLeastOnce(current, fileFilters)
647                 : noFilters || Filters.matchAtLeastOnce(current, fileFilters);
648         final String logPrefix = (addFile ? "Accepted " : "Rejected ")
649                 + (current.isDirectory() ? "directory" : "file") + " [";
650 
651         if (addFile) {
652             toPopulate.add(current);
653         }
654 
655         if (log.isDebugEnabled()) {
656             log.debug(logPrefix + getCanonicalPath(current) + "]");
657         }
658     }
659 
660     private static void validateFileOrDirectoryName(final File fileOrDir) {
661 
662         if (Os.isFamily(Os.FAMILY_WINDOWS) && !FileUtils.isValidWindowsFileName(fileOrDir)) {
663             throw new IllegalArgumentException(
664                     "The file (" + fileOrDir + ") cannot contain any of the following characters: \n"
665                             + StringUtils.join(INVALID_CHARACTERS_FOR_WINDOWS_FILE_NAME, " "));
666         }
667     }
668 
669     private static void recurseAndPopulate(final List<File> toPopulate,
670                                            final List<Filter<File>> fileFilters,
671                                            final File aDirectory,
672                                            final boolean excludeOperation,
673                                            final Log log) {
674 
675         final List<File> files = listFiles(aDirectory, fileFilters, excludeOperation, log);
676         for (File current : files) {
677             if (EXISTING_FILE.accept(current)) {
678                 toPopulate.add(current);
679             }
680 
681             if (EXISTING_DIRECTORY.accept(current)) {
682                 recurseAndPopulate(toPopulate, fileFilters, current, excludeOperation, log);
683             }
684         }
685     }
