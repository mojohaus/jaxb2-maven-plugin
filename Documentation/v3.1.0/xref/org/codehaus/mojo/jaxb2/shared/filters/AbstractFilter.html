22  import org.apache.maven.plugin.logging.Log;
23  import org.codehaus.mojo.jaxb2.AbstractJaxbMojo;
24  import org.codehaus.mojo.jaxb2.shared.Validate;
25  
26  import java.util.ArrayList;
27  import java.util.List;
28  
29  /**
30   * Abstract Filter implementation which handles separating {@code null} candidate values from non-null
31   * ones, and delegates processing to concrete subclass implementations. Also, this AbstractFilter
32   * implementation provides a standard for emitting Filter debug statements (i.e. for toString() calls).
33   *
34   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
35   * @since 2.0
36   */
37  public abstract class AbstractFilter<T> implements Filter<T> {
38  
39      /**
40       * Initial-line indent for human-readable toString() rendering.
41       */
42      protected static final String TOSTRING_INDENT = "|        ";
43  
44      class DelayedLogMessage {
45          String message;
46          String logLevel;
47  
48          public DelayedLogMessage(final String logLevel, final String message) {
49              this.logLevel = logLevel;
50              this.message = message;
51          }
52      }
53  
54      // Internal state
55      protected Log log;
56      private List<DelayedLogMessage> delayedLogMessages;
57      private boolean processNullValues;
58  
59      /**
60       * Convenience constructor which creates an AbstractFilter which <strong>does not</strong> process null values.
61       */
62      protected AbstractFilter() {
63  
64          // Assign internal state
65          delayedLogMessages = new ArrayList<DelayedLogMessage>();
66          setProcessNullValues(false);
67      }
68  
69      /**
70       * Validator method which should be called in a DI setter method to ensure that this AbstractFilter has
71       * not yet been initialized.
72       *
73       * @param setterPropertyName The name of the property to inject.
74       */
75      protected final void validateDiSetterCalledBeforeInitialization(final String setterPropertyName) {
76          Validate.isTrue(log == null, "DI Setters should only be called before initializing. Stray call: ["
77                  + setterPropertyName + "]");
78      }
79  
80      /**
81       * Adds a log message to be emitted when this AbstractFilter is initialized (and the Log is made available to it).
82       *
83       * @param logLevel The logLevel of the message to emit.
84       * @param message  The message to emit.
85       */
86      protected final void addDelayedLogMessage(final String logLevel, final String message) {
87          this.delayedLogMessages.add(new DelayedLogMessage(logLevel, message));
88      }
89  
90      /**
91       * Assigns the {@code processNullValues} property which indicates if this AbstractFilter
92       * should process null values or not.
93       *
94       * @param processNullValues {@code true} to indicate that this AbstractFilter should process null values.
95       */
96      protected final void setProcessNullValues(final boolean processNullValues) {
97  
98          validateDiSetterCalledBeforeInitialization("processNullValues");
99          this.processNullValues = processNullValues;
100     }
101 
102     /**
103      * {@inheritDoc}
104      */
105     @Override
106     public final void initialize(final Log log) {
107 
108         // Check sanity
109         Validate.notNull(log, "log");
110 
111         // Assign internal state
112         this.log = log;
113 
114         if (delayedLogMessages.size() > 0) {
115             for (DelayedLogMessage current : delayedLogMessages) {
116                 if (current.logLevel.equalsIgnoreCase("warn") && log.isWarnEnabled()) {
117                     log.warn(current.message);
118                 } else if (current.logLevel.equals("info") && log.isInfoEnabled()) {
119                     log.info(current.message);
120                 } else if (log.isDebugEnabled()) {
121                     log.debug(current.message);
122                 }
123             }
124 
125             delayedLogMessages.clear();
126         }
127 
128         // Delegate
129         onInitialize();
130     }
131 
132     /**
133      * {@inheritDoc}
134      */
135     @Override
136     public boolean isInitialized() {
137         return log != null;
138     }
139 
140     /**
141      * Override this method to perform some custom action after standard initialization is complete.
142      * Default implementation does nothing, but the log is non-null and ready for use.
143      */
144     protected void onInitialize() {
145         // Do nothing
146     }
147 
148     /**
149      * {@inheritDoc}
150      */
151     @Override
152     public final boolean accept(final T candidate) throws IllegalStateException {
153 
154         // Check sanity
155         if (log == null) {
156             throw new IllegalStateException("Filter [" + getClass().getSimpleName() + "] not initialized before use.");
157         }
158 
159         // Should we process null values?
160         if (candidate == null) {
161 
162             boolean toReturn = false;
163             if (processNullValues) {
164                 toReturn = onNullCandidate();
165             } else {
166                 if (log.isDebugEnabled()) {
167                     log.debug("Received null candidate, and Filter [" + getClass().getSimpleName()
168                             + "] is configured not to match nulls.");
169                 }
170             }
171 
172             // All done.
173             return toReturn;
174         }
175 
176         // Delegate processing.
177         return onCandidate(candidate);
178     }
179 
180     /**
181      * Override this to perform an action other than debug logging whenever this AbstractFilter
182      * receives a null candidate. You also need to override this implementation if you want to return a value other
183      * than {@code false}. There is no point in calling {@code super.onNullCandidate()} whenever this method is
184      * overridden.
185      *
186      * @return {@code false} - implying that AbstractFilters will not matchAtLeastOnce {@code null}s by default.
187      */
188     protected boolean onNullCandidate() {
189         if (log.isDebugEnabled()) {
190             log.debug("Filter [" + getClass().getSimpleName()
191                     + "] is configured to matchAtLeastOnce processing null candidate "
192                     + "values, but no implementation is supplied.");
193         }
194 
195         // Don't matchAtLeastOnce null values by default.
196         return false;
197     }
198 
199     /**
200      * <p>Method that is invoked to determine if a candidate instance should be accepted or not.</p>
201      *
202      * @param nonNullCandidate The candidate that should be tested for acceptance by this Filter. Never null.
203      * @return {@code true} if the candidate is accepted by this Filter and {@code false} otherwise.
204      */
205     protected abstract boolean onCandidate(final T nonNullCandidate);
206 
207     /**
208      * {@inheritDoc}
209      */
210     @Override
211     public String toString() {
212         return "Filter [" + getClass().getSimpleName() + "]" + AbstractJaxbMojo.NEWLINE
213                 + TOSTRING_INDENT + "Processes nulls: [" + processNullValues + "]" + AbstractJaxbMojo.NEWLINE;
214     }
215 }
