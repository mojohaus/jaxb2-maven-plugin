409     /**
410      * Acquires all municipalities within the supplied County.
411      *
412      * @param county The non-null county for which all Municipalities should be retrieved.
413      * @return A SortedSet containing the Municipalities within the supplied County.
414      */
415     public static SortedSet<Municipality> getMunicipalitiesIn(final County county) {
416 
417         // Check sanity
418         Validate.notNull(county, "Cannot handle null 'county' argument.");
419 
420         // Find all Municipalities within the supplied county.
421         final SortedSet<Municipality> toReturn = new TreeSet<Municipality>();
422         for (Municipality current : values()) {
423             if (current.getCounty() == county) {
424                 toReturn.add(current);
425             }
426         }
427 
428         // All done.
429         return toReturn;
430     }
