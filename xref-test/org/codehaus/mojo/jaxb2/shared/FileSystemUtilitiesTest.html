202     @Test
203     public void validateResolvingFilesAndRemovingExclusions() {
204 
205         // Assemble
206         final URL fileFilterDirUrl = getClass().getClassLoader().getResource("testdata/shared/filefilter");
207         final File fileFilterDir = new File(fileFilterDirUrl.getPath());
208         final File depsPropertiesFile = new File(fileFilterDir.getParentFile(), "deps1.properties");
209 
210         final List<File> fileList = new ArrayList<File>();
211         fileList.add(fsUtilitiesDirectory);
212         fileList.add(fileFilterDir);
213         fileList.add(depsPropertiesFile);
214 
215         final List<Filter<File>> matchTextFiles = PatternFileFilter.createIncludeFilterList(log, "\\.txt");
216         final List<Filter<File>> matchXmlFiles = PatternFileFilter.createIncludeFilterList(log, "\\.xml");
217         final List<Filter<File>> matchNonexistentFiles = PatternFileFilter.createIncludeFilterList(
218                 log, "\\.nonexistent");
219 
220         // Act
221         final List<File> noTextFiles = FileSystemUtilities.resolveRecursively(fileList, matchTextFiles, log);
222         final List<File> noXmlFiles = FileSystemUtilities.resolveRecursively(fileList, matchXmlFiles, log);
223         final List<File> allFiles = FileSystemUtilities.resolveRecursively(fileList, matchNonexistentFiles, log);
224         final List<File> noFilterFiles = FileSystemUtilities.resolveRecursively(fileList, null, log);
225 
226         // Assert
227         Assert.assertEquals(2, noTextFiles.size());
228         Assert.assertEquals(4, noXmlFiles.size());
229         Assert.assertEquals(5, allFiles.size());
230         Assert.assertEquals(5, noFilterFiles.size());
231 
232         final List<String> canonicalPathsForExplicitlyAddedFiles = getRelativeCanonicalPaths(
233                 noFilterFiles, depsPropertiesFile.getParentFile());
234         Assert.assertTrue(canonicalPathsForExplicitlyAddedFiles.contains("/deps1.properties"));
235 
236         final List<String> canonicalPathsForNoTextFiles = getRelativeCanonicalPaths(
237                 noTextFiles, depsPropertiesFile.getParentFile());
238         Assert.assertTrue(canonicalPathsForNoTextFiles.contains("/deps1.properties"));
239         Assert.assertTrue(canonicalPathsForNoTextFiles.contains("/filefilter/exclusion/AnXmlFile.xml"));
240 
241         final List<String> canonicalPathsForNoXmlFiles = getRelativeCanonicalPaths(
242                 noXmlFiles, depsPropertiesFile.getParentFile());
243         Assert.assertTrue(canonicalPathsForNoXmlFiles.contains("/deps1.properties"));
244         Assert.assertTrue(canonicalPathsForNoXmlFiles.contains("/filefilter/exclusion/TextFile.txt"));
245         Assert.assertTrue(canonicalPathsForNoXmlFiles.contains("/filesystemutilities/TestFile2.txt"));
246         Assert.assertTrue(canonicalPathsForNoXmlFiles.contains("/filesystemutilities/canonicals/TestFile1.txt"));
247 
248         final List<String> canonicalPathsForAllFiles = getRelativeCanonicalPaths(
249                 allFiles, depsPropertiesFile.getParentFile());
250         Assert.assertTrue(canonicalPathsForAllFiles.contains("/deps1.properties"));
251         Assert.assertTrue(canonicalPathsForAllFiles.contains("/filefilter/exclusion/TextFile.txt"));
252         Assert.assertTrue(canonicalPathsForAllFiles.contains("/filesystemutilities/TestFile2.txt"));
253         Assert.assertTrue(canonicalPathsForAllFiles.contains("/filesystemutilities/canonicals/TestFile1.txt"));
254         Assert.assertTrue(canonicalPathsForAllFiles.contains("/filefilter/exclusion/AnXmlFile.xml"));
255     }

406     private List<String> getRelativeCanonicalPaths(final List<File> fileList, final File cutoff) {
407 
408         final String cutoffPath = FileSystemUtilities.getCanonicalPath(cutoff).replace(File.separator, "/");
409         final List<String> toReturn = new ArrayList<String>();
410 
411         for (File current : fileList) {
412 
413             final String canPath = FileSystemUtilities.getCanonicalPath(current).replace(File.separator, "/");
414             if (!canPath.startsWith(cutoffPath)) {
415                 throw new IllegalArgumentException("Illegal cutoff provided. Cutoff: [" + cutoffPath
416                         + "] must be a parent to CanonicalPath [" + canPath + "]");
417             }
418 
419             toReturn.add(canPath.substring(cutoffPath.length()));
420         }
421         Collections.sort(toReturn);
422 
423         return toReturn;
424     }
