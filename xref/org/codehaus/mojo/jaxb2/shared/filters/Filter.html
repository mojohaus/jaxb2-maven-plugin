1   package org.codehaus.mojo.jaxb2.shared.filters;
2   
3   import org.apache.maven.plugin.logging.Log;
4   
5   /**
6    * Generic Filter specification, whose implementations define if candidate objects should be accepted or not.
7    *
8    * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
9    * @since 2.0
10   */
11  public interface Filter<T> {
12  
13      /**
14       * Initializes this Filter, and assigns the supplied Log for use by this Filter.
15       *
16       * @param log The non-null Log which should be used by this Filter to emit log messages.
17       */
18      void initialize(Log log);
19  
20      /**
21       * @return {@code true} if this Filter has been properly initialized (by a call to the {@code initialize} method).
22       */
23      boolean isInitialized();
24  
25      /**
26       * <p>Method that is invoked to determine if a candidate instance should be accepted or not.
27       * Implementing classes should be prepared to handle {@code null} candidate objects.</p>
28       *
29       * @param candidate The candidate that should be tested for acceptance by this Filter.
30       * @return {@code true} if the candidate is accepted by this Filter and {@code false} otherwise.
31       * @throws java.lang.IllegalStateException if this Filter is not initialized by a call to the
32       *                                         initialize method before calling this matchAtLeastOnce method.
33       */
34      boolean accept(T candidate) throws IllegalStateException;
35  }
