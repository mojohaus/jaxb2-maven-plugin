23      @Test
24      public void validateDomHelperAccessors() throws Exception {
25  
26          // Assemble
27          final String xsd = PropertyResources.readFully("testdata/schemageneration/javadoc/enums/rawEnumSchema.xsd");
28          final Document document = XsdGeneratorHelper.parseXmlStream(new StringReader(xsd));
29  
30          final Element schemaElement = document.getDocumentElement();
31          final Element simpleTypeElement = getChildOf(schemaElement, "simpleType", "foodPreference");
32          final Element restrictionElement = getChildOf(simpleTypeElement, "restriction", null);
33  
34          // Act
35          final List<Element> enumElements = new ArrayList<Element>();
36          final NodeList childNodes = restrictionElement.getChildNodes();
37          for(int i = 0; i < childNodes.getLength(); i++) {
38  
39              final Node current = childNodes.item(i);
40              if(current.getNodeType() == Node.ELEMENT_NODE && current.getLocalName().equals("enumeration")) {
41                  enumElements.add((Element) current);
42              }
43          }
44  
45          final Map<String, String> xpath2ValueMap = new TreeMap<String, String>();
46          for(Element current : enumElements) {
47              final String currentXPath = DomHelper.getXPathFor(current);
48              xpath2ValueMap.put(currentXPath, DomHelper.getValueAttribute(current));
49          }
50  
51          // Assert
52          Assert.assertNotNull(xpath2ValueMap);
53          Assert.assertEquals(3, xpath2ValueMap.size());
54  
55          final String prefix ="#document/xs:schema/xs:simpleType[@name='foodPreference']/" +
56                  "xs:restriction/xs:enumeration[@value='";
57  
58          Assert.assertEquals("LACTO_VEGETARIAN", xpath2ValueMap.get(prefix + "LACTO_VEGETARIAN']"));
59          Assert.assertEquals("NONE", xpath2ValueMap.get(prefix + "NONE']"));
60          Assert.assertEquals("VEGAN", xpath2ValueMap.get(prefix + "VEGAN']"));
61      }
62

66  
67      private Element getChildOf(final Element parent, final String localName, final String name) {
68  
69          final NodeList children = parent.getChildNodes();
70          for (int i = 0; i < children.getLength(); i++) {
71  
72              Node current = children.item(i);
73              if (current.getNodeType() == Node.ELEMENT_NODE) {
74  
75                  final String currentLocalName = current.getLocalName();
76                  final String nameAttribute = DomHelper.getNameAttribute(current);
77  
78                  if(currentLocalName.equals(localName) && (name == null || name.equals(nameAttribute))) {
79                      return (Element) current;
80                  }
81              }
82          }
83  
84          return null;
85      }
