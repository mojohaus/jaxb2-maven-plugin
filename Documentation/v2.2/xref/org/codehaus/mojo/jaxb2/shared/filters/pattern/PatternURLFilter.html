1   package org.codehaus.mojo.jaxb2.shared.filters.pattern;
2   
3   import java.net.URL;
4   import java.util.List;
5   
6   /**
7    * AbstractPatternFilter matching the string of URLs with to a set of Regular expressions.
8    *
9    * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
10   * @since 2.0
11   */
12  public class PatternURLFilter extends AbstractPatternFilter<URL> {
13  
14      /**
15       * Converter returning each URL's {@code toString()} form, after normalizing it, using the
16       * algorithm {@code toConvert.toURI().normalize().toURL().toString();}
17       */
18      public static final StringConverter<URL> NORMALIZED_URL_CONVERTER = new StringConverter<URL>() {
19  
20          /**
21           * {@inheritDoc}
22           */
23          @Override
24          public String convert(final URL toConvert) {
25              try {
26                  return toConvert.toURI().normalize().toURL().toString();
27              } catch (Exception e) {
28  
29                  // This should really not happen.
30                  throw new IllegalArgumentException("Could not handle URL [" + toConvert + "]", e);
31              }
32          }
33      };
34  
35      /**
36       * Creates a new ExclusionRegularExpressionURLFilter using the supplied patternStrings which are
37       * matched against each full - normalized - URL. The {@code NORMALIZED_URL_CONVERTER} is used to convert
38       * URLs to strings.
39       *
40       * @param patternStrings The list of patternStrings to be used as regular expression matchers against the
41       *                       normalized URLs.
42       * @see #NORMALIZED_URL_CONVERTER
43       * @see #convert(java.util.List, String)
44       */
45      public PatternURLFilter(final List<String> patternStrings) {
46          this(false, "", patternStrings, NORMALIZED_URL_CONVERTER, true);
47      }
48  
49      /**
50       * Compound constructor creating an ExclusionRegularExpressionURLFilter from the supplied parameters.
51       *
52       * @param processNullValues             if {@code true}, this ExclusionRegularExpressionURLFilter process null
53       *                                      candidate values.
54       * @param patternPrefix                 a prefix to be prepended to any patterns submitted to
55       *                                      this PatternURLFilter.
56       * @param patterns                      The non-null list of Patters which should be applied within this
57       *                                      ExclusionRegularExpressionURLFilter. A candidate of type T should only
58       *                                      be accepted by this ExclusionRegularExpressionURLFilter if all supplied
59       *                                      patterns matchAtLeastOnce the candidate.
60       * @param converter                     The StringConverter used to convert T-type objects to Strings which should
61       *                                      be matched by all supplied Patterns to T-object candidates.
62       * @param acceptCandidateOnPatternMatch if {@code true}, this ExclusionRegularExpressionURLFilter will matchAtLeastOnce
63       *                                      candidate objects that match at least one of the supplied patterns. if
64       *                                      {@code false}, this ExclusionRegularExpressionURLFilter will noFilterMatches
65       *                                      candidates that match at least one of the supplied patterns.
66       */
67      public PatternURLFilter(final boolean processNullValues,
68                              final String patternPrefix,
69                              final List<String> patterns,
70                              final StringConverter<URL> converter,
71                              final boolean acceptCandidateOnPatternMatch) {
72          super();
73  
74          // Assign internal state
75          setProcessNullValues(processNullValues);
76          setAcceptCandidateOnPatternMatch(acceptCandidateOnPatternMatch);
77          setPatternPrefix(patternPrefix);
78          setPatterns(patterns);
79          setConverter(converter);
80      }
81  }
