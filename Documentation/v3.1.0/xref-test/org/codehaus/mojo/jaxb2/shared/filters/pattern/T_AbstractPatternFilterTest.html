17      private static final StringConverter<String> UNITY_STRING_CONVERTER = new StringConverter<String>() {
18          @Override
19          public String convert(final String toConvert) {
20              return toConvert;
21          }

45      @Test
46      public void validateNoAcceptedResultsWhenProcessingMessagesOnNullFilters() {
47  
48          // Assemble
49          final boolean processNullValues = false;
50          final List<String> patterns = null;
51          final boolean acceptCandidateOnPatternMatch = true;
52  
53          final DebugPatternFilter unitUnderTest = new DebugPatternFilter(
54                  processNullValues,
55                  patterns,
56                  UNITY_STRING_CONVERTER,
57                  acceptCandidateOnPatternMatch);
58  
59          // Act
60          unitUnderTest.initialize(log);
61          unitUnderTest.accept("first");
62          unitUnderTest.accept(null);
63          unitUnderTest.accept("third");
64  
65          // Assert
66          final SortedMap<Integer, String[]> invocations = unitUnderTest.invocations;
67          Assert.assertEquals(2, invocations.size());
68          Assert.assertEquals("first", invocations.get(1)[0]);
69          Assert.assertEquals(false, Boolean.parseBoolean(invocations.get(1)[1]));
70          Assert.assertEquals("third", invocations.get(2)[0]);
71          Assert.assertEquals(false, Boolean.parseBoolean(invocations.get(2)[1]));
72      }
73  
74      @Test
75      public void validateAcceptingFilterMessages() {
76  
77          // Assemble
78          final boolean processNullValues = false;
79          final boolean acceptCandidateOnPatternMatch = true;
80          final List<String> patterns = Arrays.asList("f.*t"); // Should match 'first' but not 'third'

86                  acceptCandidateOnPatternMatch);

96          Assert.assertEquals(2, invocations.size());
97          Assert.assertEquals("first", invocations.get(1)[0]);
98          Assert.assertEquals(true, Boolean.parseBoolean(invocations.get(1)[1]));
99          Assert.assertEquals("third", invocations.get(2)[0]);
100         Assert.assertEquals(false, Boolean.parseBoolean(invocations.get(2)[1]));
