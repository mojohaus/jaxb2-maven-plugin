14  public class LocaleFacetTest {
15  
16      // Shared state
17      private static final Locale FRENCH_LOCALE = Locale.FRENCH;
18      private Locale defaultLocale;
19      private BufferingLog log;
20  
21      @Before
22      public void setupSharedState() {
23          this.log = new BufferingLog(BufferingLog.LogLevel.DEBUG);
24          defaultLocale = Locale.getDefault();
25      }
26  
27      @Test
28      public void validateLocaleParsing() throws MojoExecutionException {
29  
30          // Assemble
31          final String frenchLocaleString = "fr";
32  
33          // Act & Assert
34          final LocaleFacet facet = LocaleFacet.createFor(frenchLocaleString, log);
35          Assert.assertEquals(defaultLocale, Locale.getDefault());
36  
37          facet.setup();
38          Assert.assertEquals(FRENCH_LOCALE.toString(), Locale.getDefault().toString());
39  
40          facet.restore();
41          Assert.assertEquals(defaultLocale.toString(), Locale.getDefault().toString());
42      }
43  
44      @Test(expected = MojoExecutionException.class)
45      public void validateExceptionOnIncorrectLocaleString() throws MojoExecutionException {
46  
47          // Act & Assert
48          LocaleFacet.createFor("not,a,properly,formatted,locale_string", log);
49      }
50  
51      @Test
52      public void validateNoExceptionOnUnknownLocaleString() throws MojoExecutionException {
53  
54          // Act & Assert
55          LocaleFacet.createFor("thisIsAnUnknownLocale", log);
56      }
57  }
