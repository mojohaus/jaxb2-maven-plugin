42  /**
43   * <p>Mojo that creates XML schema(s) from compile-scope Java sources or binaries
44   * by invoking the JAXB SchemaGenerator. This implementation is tailored to use the
45   * JAXB Reference Implementation from project Kenai.</p>
46   * <p>Note that the SchemaGenerationMojo was completely re-implemented for the 2.x versions.
47   * Its configuration semantics and parameter set is <strong>not necessarily
48   * backwards compatible</strong> with the 1.x plugin versions. If you are
49   * upgrading from version 1.x of the plugin, read the documentation carefully.</p>
50   *
51   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>
52   * @see <a href="https://jaxb.java.net/">The JAXB Reference Implementation</a>
53   */
54  @Mojo(name = "schemagen",
55          defaultPhase = LifecyclePhase.GENERATE_RESOURCES,
56          requiresDependencyResolution = ResolutionScope.COMPILE,
57          threadSafe = true)
58  public class SchemaGenerationMojo extends AbstractXsdGeneratorMojo {
59  
60      /**
61       * The last part of the stale fileName for this SchemaGenerationMojo.
62       */
63      public static final String STALE_FILENAME = "schemaGenerationStaleFlag";
64  
65      /**
66       * <p>Default exclude file name suffixes for sources, used unless overridden by an
67       * explicit configuration in the {@code sourceExcludeSuffixes} parameter.
68       * The default values are found as follows:</p>
69       * <pre>
70       *     <code>
71       *         final List&lt;Filter&lt;File&gt;&gt; srcTemp = new ArrayList&lt;Filter&lt;File&gt;&gt;();
72       *         srcTemp.addAll(AbstractJaxbMojo.STANDARD_EXCLUDE_FILTERS);
73       *         srcTemp.add(new PatternFileFilter(Arrays.asList("\\.xjb", "\\.xsd", "\\.properties"), true));
74       *         STANDARD_SOURCE_EXCLUDE_FILTERS = Collections.unmodifiableList(srcTemp);
75       *     </code>
76       * </pre>
77       *
78       * @see #STANDARD_EXCLUDE_FILTERS
79       */
80      public static final List<Filter<File>> STANDARD_SOURCE_EXCLUDE_FILTERS;
81  
82      static {
83  
84          final List<Filter<File>> srcTemp = new ArrayList<Filter<File>>();
85          srcTemp.addAll(AbstractJaxbMojo.STANDARD_EXCLUDE_FILTERS);
86          srcTemp.add(new PatternFileFilter(Arrays.asList("\\.xjb", "\\.xsd", "\\.properties"), true));
87          STANDARD_SOURCE_EXCLUDE_FILTERS = Collections.unmodifiableList(srcTemp);
88      }
89  
90      /**
91       * <p>Parameter holding List of paths to files and/or directories which should be recursively searched
92       * for Java source files. Only files or directories that actually exist will be included (in the case of files)
93       * or recursively searched for source files to include (in the case of directories or JARs).
94       * Configure using standard Maven structure for Lists:</p>
95       * <pre>
96       * <code>
97       *   &lt;configuration>
98       *   ...
99       *       &lt;sources>
100      *          &lt;source>/a/full/absolute/path/to/a/SourceFile.java&lt;/source>
101      *          &lt;source>target/some/sourceJar.jar&lt;/source>
102      *          &lt;source>src/main/java&lt;/source>
103      *      &lt;/sources>
104      *   &lt;/configuration>
105      * </code>
106      * </pre>
107      * <p><strong>Note</strong>: if configured, the sources parameters replace the default
108      * value, which is a List containing the paths to the directories defined by
109      * {@code getProject().getCompileSourceRoots()}.</p>
110      *
111      * @since 2.0
112      */
113     @Parameter(required = false)
114     private List<String> sources;
115 
116     /**
117      * <p>Parameter holding a List of Filters, used to match all files under the {@code sources} directories
118      * which should <strong>not</strong> be considered SchemaGenerator source files. (The filters identify files to
119      * exclude, and hence this parameter is called {@code schemaSourceExcludeFilters}). If a file under any of the
120      * source directories matches at least one of the Filters supplied in the {@code schemaSourceExcludeFilters},
121      * it is not considered an XJC source file, and therefore excluded from processing.</p>
122      * <p>If not explicitly provided, the Mojo uses the value within {@code STANDARD_SOURCE_EXCLUDE_FILTERS}.
123      * The algorithm for finding XJC sources is as follows:</p>
124      * <ol>
125      * <li>Find all files given in the sources List. Any Directories provided are searched for files
126      * recursively.</li>
127      * <li>Exclude any found files matching any of the supplied {@code schemaSourceExcludeFilters} List.</li>
128      * <li>The remaining Files are submitted for processing by the XJC tool.</li>
129      * </ol>
130      * <p><strong>Example:</strong> The following configuration would exclude any sources whose names end with
131      * {@code .txt} or {@code .foo}:</p>
132      * <pre>
133      *     <code>
134      *         &lt;configuration>
135      *         ...
136      *              &lt;schemaSourceExcludeFilters>
137      *                  &lt;filter implementation="org.codehaus.mojo.jaxb2.shared.filters.pattern.PatternFileFilter">
138      *                      &lt;patterns>
139      *                          &lt;pattern>\.txt&lt;/pattern>
140      *                          &lt;pattern>\.foo&lt;/pattern>
141      *                      &lt;/patterns>
142      *                  &lt;/filter>
143      *              &lt;/schemaSourceExcludeFilters>
144      *         &lt;/configuration>
145      *     </code>
146      * </pre>
147      * <p>Note that inner workings of the Dependency Injection mechanism used by Maven Plugins (i.e. the DI from
148      * the Plexus container) requires that the full class name to the Filter implementation should be supplied for
149      * each filter, as is illustrated in the sample above. This is true also if you implement custom Filters.</p>
150      *
151      * @see #STANDARD_SOURCE_EXCLUDE_FILTERS
152      * @see org.codehaus.mojo.jaxb2.shared.filters.pattern.PatternFileFilter
153      * @see org.codehaus.mojo.jaxb2.shared.filters.pattern.AbstractPatternFilter
154      * @see org.codehaus.mojo.jaxb2.shared.filters.AbstractFilter
155      * @since 2.0
156      */
157     @Parameter(required = false)
158     private List<Filter<File>> schemaSourceExcludeFilters;
159 
160     /**
161      * <p>The directory where the generated XML Schema file(s) will be
162      * placed, after all transformations are done.</p>
163      */
164     @Parameter(defaultValue = "${project.build.directory}/generated-resources/schemagen", required = true)
165     private File outputDirectory;
166 
167     /**
168      * <p>The directory where the {@code schemagen} tool will output XSDs, episode files - and intermediary bytecode
169      * files. From this directory the XSDs and the episode file (but not the bytecode files) will be copied to the
170      * outputDirectory for further processing.</p>
171      *
172      * @see #outputDirectory
173      */
174     @Parameter(defaultValue = "${project.build.directory}/schemagen-work/compile_scope", required = true)
175     private File workDirectory;
176 
177     /**
178      * {@inheritDoc}
179      */
180     @Override
181     protected List<URL> getCompiledClassNames() {
182 
183         List<Filter<File>> excludeFilters = schemaSourceExcludeFilters == null
184                 ? STANDARD_BYTECODE_EXCLUDE_FILTERS
185                 : schemaSourceExcludeFilters;
186         Filters.initialize(getLog(), excludeFilters);
187 
188         try {
189             return FileSystemUtilities.filterFiles(
190                     getProject().getBasedir(),
191                     null,
192                     getProject().getCompileClasspathElements(),
193                     getLog(),
194                     "compiled bytecode",
195                     excludeFilters);
196         } catch (DependencyResolutionRequiredException e) {
197             throw new IllegalStateException("Could not resolve dependencies.", e);
198         }
199     }
200 
201     /**
202      * {@inheritDoc}
203      */
204     @Override
205     protected List<URL> getSources() {
206 
207         final List<Filter<File>> sourceExcludes = schemaSourceExcludeFilters == null
208                 ? STANDARD_SOURCE_EXCLUDE_FILTERS
209                 : schemaSourceExcludeFilters;
210         Filters.initialize(getLog(), sourceExcludes);
211 
212         final List<String> defaultSources = getProject().getCompileSourceRoots();
213 
214         // All Done.
215         return FileSystemUtilities.filterFiles(
216                 getProject().getBasedir(),
217                 (sources == null) ? defaultSources : sources,
218                 defaultSources,
219                 getLog(),
220                 "sources",
221                 sourceExcludes);
222     }
223

227     @Override

230     }

234      */
235     @Override
236     protected List<String> getClasspath() throws MojoExecutionException {
237 
238         List<String> toReturn = null;
239         try {
240             toReturn = getProject().getCompileClasspathElements();
241         } catch (DependencyResolutionRequiredException e) {
242             throw new MojoExecutionException("Could not acquire compile classpath elements from MavenProject", e);
243         }
244 
245         // All done.
246         return toReturn;
247     }

254         return STALE_FILENAME;

260     @Override
261     protected File getWorkDirectory() {

263     }
264 
265     /**
266      * {@inheritDoc}
267      */
268     @Override
269     protected void addResource(final Resource resource) {
270 
271         if(resource != null) {
272 
273             final String newDirectory = resource.getDirectory();
274 
275             // Is the supplied resource already added?
276             final List<Resource> currentResources = getProject().getResources();
277 
278             if(getLog().isDebugEnabled()) {
279                 getLog().debug("Candidate Resource Directory [" + newDirectory + "]");
280                 getLog().debug("Found [" + currentResources.size() + "] current Resources: "
281                         + currentResources);
282             }
283 
284             for (Resource current : currentResources) {
285 
286                 // Is the resource already added?
287                 if(current.getDirectory() != null && current.getDirectory().equalsIgnoreCase(newDirectory)) {
288                     getLog().debug("Resource already added [" + newDirectory + "]. Not adding again.");
289                     return;
290                 }
291             }
292 
293             // Add the new Resource
294             currentResources.add(resource);
295 
296             if(getLog().isDebugEnabled()) {
297                 getLog().debug("Added resource [" + newDirectory + "] to existing resources.");
298             }
299         }
300     }
