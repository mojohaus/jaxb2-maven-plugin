3   import org.codehaus.mojo.jaxb2.BufferingLog;
4   import org.codehaus.mojo.jaxb2.shared.FileSystemUtilities;
5   import org.junit.Assert;
6   import org.junit.Before;
7   
8   import java.io.File;
9   import java.net.URL;
10  import java.util.Map;
11  import java.util.TreeMap;
12  
13  /**
14   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
15   */
16  public abstract class AbstractPatternFilterTest {
17  
18      // Shared state
19      protected File exclusionDirectory;
20      protected File[] fileList;
21      protected BufferingLog log;
22      private String defaultExclusionDirectory;
23  
24      public AbstractPatternFilterTest(final String defaultExclusionDirectory) {
25          this.defaultExclusionDirectory = defaultExclusionDirectory;
26      }
27  
28      public AbstractPatternFilterTest() {
29          this("testdata/shared/filefilter/exclusion");
30      }
31  
32      @Before
33      @SuppressWarnings("all")
34      public void setupSharedState() {
35  
36          log = new BufferingLog(BufferingLog.LogLevel.DEBUG);
37  
38          final URL exclusionDirUrl = getClass().getClassLoader().getResource(defaultExclusionDirectory);
39          exclusionDirectory = new File(exclusionDirUrl.getPath());
40          Assert.assertTrue(FileSystemUtilities.EXISTING_DIRECTORY.accept(exclusionDirectory));
41  
42          fileList = exclusionDirectory.listFiles();
43          for (File current : fileList) {
44              Assert.assertTrue(FileSystemUtilities.EXISTING_FILE.accept(current));
45          }
46  
47          // Delegate
48          onSetup();
49      }
50  
51      protected void onSetup() {
52          // Do nothing
53      }
54  
55      protected <T> Map<String, Boolean> applyFilterAndRetrieveResults(final AbstractPatternFilter<T> unitUnderTest,
56                                                                       final T[] candidates,
57                                                                       final StringConverter<T> converter) {
58  
59          Map<String, Boolean> toReturn = new TreeMap<String, Boolean>();
60  
61          for (T current : candidates) {
62              toReturn.put(converter.convert(current), unitUnderTest.accept(current));
63          }
64  
65          return toReturn;
66      }
67  }
