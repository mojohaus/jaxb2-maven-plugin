22  import java.net.URL;
23  import java.util.List;
24  
25  /**
26   * AbstractPatternFilter matching the string of URLs with to a set of Regular expressions.
27   *
28   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
29   * @since 2.0
30   */
31  public class PatternURLFilter extends AbstractPatternFilter<URL> {
32  
33      /**
34       * Converter returning each URL's {@code toString()} form, after normalizing it, using the
35       * algorithm {@code toConvert.toURI().normalize().toURL().toString();}
36       */
37      @SuppressWarnings("all")
38      public static final StringConverter<URL> NORMALIZED_URL_CONVERTER = new StringConverter<URL>() {
39  
40          /**
41           * {@inheritDoc}
42           */
43          @Override
44          public String convert(final URL toConvert) {
45              try {
46                  return toConvert.toURI().normalize().toURL().toString();
47              } catch (Exception e) {
48  
49                  // This should really not happen.
50                  throw new IllegalArgumentException("Could not handle URL [" + toConvert + "]", e);
51              }
52          }
53      };
54  
55      /**
56       * Creates a new ExclusionRegularExpressionURLFilter using the supplied patternStrings which are
57       * matched against each full - normalized - URL. The {@code NORMALIZED_URL_CONVERTER} is used to convert
58       * URLs to strings.
59       *
60       * @param patternStrings The list of patternStrings to be used as regular expression matchers against the
61       *                       normalized URLs.
62       * @see #NORMALIZED_URL_CONVERTER
63       * @see #convert(java.util.List, String)
64       */
65      public PatternURLFilter(final List<String> patternStrings) {
66          this(false, "", patternStrings, NORMALIZED_URL_CONVERTER, true);
67      }
68  
69      /**
70       * Compound constructor creating an ExclusionRegularExpressionURLFilter from the supplied parameters.
71       *
72       * @param processNullValues             if {@code true}, this ExclusionRegularExpressionURLFilter process null
73       *                                      candidate values.
74       * @param patternPrefix                 a prefix to be prepended to any patterns submitted to
75       *                                      this PatternURLFilter.
76       * @param patterns                      The non-null list of Patters which should be applied within this
77       *                                      ExclusionRegularExpressionURLFilter. A candidate of type T should only
78       *                                      be accepted by this ExclusionRegularExpressionURLFilter if all supplied
79       *                                      patterns matchAtLeastOnce the candidate.
80       * @param converter                     The StringConverter used to convert T-type objects to Strings which should
81       *                                      be matched by all supplied Patterns to T-object candidates.
82       * @param acceptCandidateOnPatternMatch if {@code true}, this ExclusionRegularExpressionURLFilter will matchAtLeastOnce
83       *                                      candidate objects that match at least one of the supplied patterns. if
84       *                                      {@code false}, this ExclusionRegularExpressionURLFilter will noFilterMatches
85       *                                      candidates that match at least one of the supplied patterns.
86       */
87      public PatternURLFilter(final boolean processNullValues,
88                              final String patternPrefix,
89                              final List<String> patterns,
90                              final StringConverter<URL> converter,
91                              final boolean acceptCandidateOnPatternMatch) {
92          super();
93  
94          // Assign internal state
95          setProcessNullValues(processNullValues);
96          setAcceptCandidateOnPatternMatch(acceptCandidateOnPatternMatch);
97          setPatternPrefix(patternPrefix);
98          setPatterns(patterns);
99          setConverter(converter);
100     }
101 }
