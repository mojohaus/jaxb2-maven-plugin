404     /**
405      * Simple helper class wrapping the path2LocationMap and the sortableLocations2JavaDocDataMap.
406      */
407     class MapWrapper {
408 
409         SortedMap<String, SortableLocation> path2LocationMap;
410         SortedMap<SortableLocation, JavaDocData> sortableLocations2JavaDocDataMap;
411 
412         /**
413          * Creates a MapWrapper using the data retrieved from a SearchableDocumentation
414          *
415          * @param searchableDocumentation A non-null SearchableDocumentation instance.
416          */
417         public MapWrapper(final SearchableDocumentation searchableDocumentation) {
418 
419             // Check sanity
420             Validate.notNull(searchableDocumentation, "searchableDocumentation");
421 
422             // Assign state
423             this.sortableLocations2JavaDocDataMap = searchableDocumentation.getAll();
424             this.path2LocationMap = new TreeMap<String, SortableLocation>();
425 
426             for (Map.Entry<SortableLocation, JavaDocData> current : sortableLocations2JavaDocDataMap.entrySet()) {
427                 path2LocationMap.put(current.getKey().getPath(), current.getKey());
428             }
429         }
430 
431         /**
432          * Validates that the JavaDoc found at the supplied SortableLocation path equals the expected value.
433          *
434          * @param expected The expected JavaDoc comment text.
435          * @param path     The SortableLocation path where the text was expected.
436          * @see SortableLocation#getPath()
437          */
438         public void validateJavaDocCommentText(final String expected, final String path) {
439 
440             final SortableLocation sortableLocation = path2LocationMap.get(path);
441             final JavaDocData xmlWrapperJavaDocData = sortableLocations2JavaDocDataMap.get(sortableLocation);
442 
443             // All Done.
444             Assert.assertEquals(expected, xmlWrapperJavaDocData.getComment());
445         }
446     }

460     private SearchableDocumentation getSearchableDocumentationFor(final JavaDocExtractor unitUnderTest,
461                                                                   final int expectedNumberOfFiles,
462                                                                   final File... sourceFileDirectories) {
463 
464         // Ensure that the encoding is correctly set
465         unitUnderTest.setEncoding("UTF-8");
466 
467         // Convert the supplied directory Files to a List
468         final List<File> sourceDirs = new ArrayList<File>();
469         Collections.addAll(sourceDirs, sourceFileDirectories);
470 
471         // Exclude any ".xsd" files found within the source directory files given
472         final List<Filter<File>> excludeFilesMatching = new ArrayList<Filter<File>>();
473         excludeFilesMatching.add(new PatternFileFilter(Collections.singletonList("\\.xsd")));
474         Filters.initialize(log, excludeFilesMatching);
475 
476         // Find all normal Files not being ".xsd" files below the supplied sourceDirs
477         final List<File> sourceFiles = FileSystemUtilities.resolveRecursively(sourceDirs, excludeFilesMatching, log);
478         Assert.assertEquals(expectedNumberOfFiles, sourceFiles.size());
479 
480         // Add the found files as source files
481         unitUnderTest.addSourceFiles(sourceFiles);
482 
483         // Launch the JavaDocExtractor and find
484         // the resulting SearchableDocumentation.
485         return unitUnderTest.process();
486     }
