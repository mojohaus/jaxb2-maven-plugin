13  /**
14   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
15   */
16  public class PatternURLFilterTest extends AbstractPatternFilterTest {
17  
18      // Shared state
19      private URL[] urlList;
20  
21      /**
22       * {@inheritDoc}
23       */
24      @Override
25      protected void onSetup() {
26  
27          urlList = new URL[fileList.length];
28  
29          for (int i = 0; i < fileList.length; i++) {
30              Assert.assertTrue(FileSystemUtilities.EXISTING_FILE.accept(fileList[i]));
31              try {
32                  urlList[i] = fileList[i].toURI().normalize().toURL();
33              } catch (MalformedURLException e) {
34                  throw new IllegalStateException("Could not populate urlList", e);
35              }
36          }
37      }
38  
39      @Test(expected = IllegalStateException.class)
40      public void validateExceptionOnNotInitializingFilterBeforeUse() {
41  
42          // Assemble
43          final PatternURLFilter unitUnderTest = new PatternURLFilter(null);
44  
45          // Act & Assert
46          unitUnderTest.accept(urlList[0]);
47      }
48  
49      @Test
50      public void validateAcceptNothingOnNullPatterns() {
51  
52          // Assemble
53          final PatternURLFilter unitUnderTest = new PatternURLFilter(null);
54  
55          // Act
56          unitUnderTest.initialize(log);
57          final Map<String, Boolean> result = applyFilterAndRetrieveResults(unitUnderTest,
58                  urlList,
59                  PatternURLFilter.NORMALIZED_URL_CONVERTER);
60  
61          // Assert
62          for (Map.Entry<String, Boolean> current : result.entrySet()) {
63              Assert.assertFalse(current.getValue());
64          }
65      }
66  
67      @Test
68      public void validateExcludingMatchingURLs() {
69  
70          // Assemble
71          final String urlProtocol = "file";
72          final List<String> txtFileSuffixExclusion = Arrays.asList(urlProtocol + ":/"
73                  + PatternFileFilter.PATTERN_LETTER_DIGIT_PUNCT + "\\.txt");
74          final PatternURLFilter unitUnderTest = new PatternURLFilter(txtFileSuffixExclusion);
75          unitUnderTest.setAcceptCandidateOnPatternMatch(false);
76          unitUnderTest.initialize(log);
77  
78          // Act
79          final Map<String, Boolean> result = applyFilterAndRetrieveResults(
80                  unitUnderTest,
81                  urlList,
82                  PatternURLFilter.NORMALIZED_URL_CONVERTER);
83  
84          // Assert
85          for (Map.Entry<String, Boolean> current : result.entrySet()) {
86              if (current.getKey().endsWith("xml")) {
87                  Assert.assertTrue(result.get(current.getKey()));
88              } else {
89                  Assert.assertFalse(result.get(current.getKey()));
90              }
91          }
92      }
93  }
