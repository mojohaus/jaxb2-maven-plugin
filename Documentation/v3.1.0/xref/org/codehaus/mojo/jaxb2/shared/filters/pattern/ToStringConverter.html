22  /**
23   * Trivial converter using the {@code toString()} method to convert a T object to a String.
24   *
25   * @author <a href="mailto:lj@jguru.se">Lennart J&ouml;relid</a>, jGuru Europe AB
26   * @since 2.0
27   */
28  public class ToStringConverter<T> implements StringConverter<T> {
29  
30      /**
31       * {@inheritDoc}
32       */
33      @Override
34      public String convert(final T toConvert) {
35          return toConvert.toString();
36      }
37  }
