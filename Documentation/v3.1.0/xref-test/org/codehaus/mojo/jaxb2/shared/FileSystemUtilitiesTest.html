243     @Test
244     public void validateResolvingFilesAndRemovingExclusions()
245     {
246 
247         // Assemble
248         final URL fileFilterDirUrl = getClass().getClassLoader().getResource( "testdata/shared/filefilter" );
249         final File fileFilterDir = new File( fileFilterDirUrl.getPath() );
250         final File depsPropertiesFile = new File( fileFilterDir.getParentFile(), "deps1.properties" );
251 
252         final List<File> fileList = new ArrayList<File>();
253         fileList.add( fsUtilitiesDirectory );
254         fileList.add( fileFilterDir );
255         fileList.add( depsPropertiesFile );
256 
257         final List<Filter<File>> matchTextFiles = PatternFileFilter.createIncludeFilterList( log, "\\.txt" );
258         final List<Filter<File>> matchXmlFiles = PatternFileFilter.createIncludeFilterList( log, "\\.xml" );
259         final List<Filter<File>> matchNonexistentFiles = PatternFileFilter.createIncludeFilterList( log,
260                 "\\.nonexistent" );
261 
262         // Act
263         final List<File> noTextFiles = FileSystemUtilities.resolveRecursively( fileList, matchTextFiles, log );
264         final List<File> noXmlFiles = FileSystemUtilities.resolveRecursively( fileList, matchXmlFiles, log );
265         final List<File> allFiles = FileSystemUtilities.resolveRecursively( fileList, matchNonexistentFiles, log );
266         final List<File> noFilterFiles = FileSystemUtilities.resolveRecursively( fileList, null, log );
267 
268         // Assert
269         Assert.assertEquals( 2, noTextFiles.size() );
270         Assert.assertEquals( 4, noXmlFiles.size() );
271         Assert.assertEquals( 5, allFiles.size() );
272         Assert.assertEquals( 5, noFilterFiles.size() );
273 
274         final List<String> canonicalPathsForExplicitlyAddedFiles = getRelativeCanonicalPaths( noFilterFiles,
275                 depsPropertiesFile.getParentFile() );
276         Assert.assertTrue( canonicalPathsForExplicitlyAddedFiles.contains( "/deps1.properties" ) );
277 
278         final List<String> canonicalPathsForNoTextFiles = getRelativeCanonicalPaths( noTextFiles,
279                 depsPropertiesFile.getParentFile() );
280         Assert.assertTrue( canonicalPathsForNoTextFiles.contains( "/deps1.properties" ) );
281         Assert.assertTrue( canonicalPathsForNoTextFiles.contains( "/filefilter/exclusion/AnXmlFile.xml" ) );
282 
283         final List<String> canonicalPathsForNoXmlFiles = getRelativeCanonicalPaths( noXmlFiles,
284                 depsPropertiesFile.getParentFile() );
285         Assert.assertTrue( canonicalPathsForNoXmlFiles.contains( "/deps1.properties" ) );
286         Assert.assertTrue( canonicalPathsForNoXmlFiles.contains( "/filefilter/exclusion/TextFile.txt" ) );
287         Assert.assertTrue( canonicalPathsForNoXmlFiles.contains( "/filesystemutilities/TestFile2.txt" ) );
288         Assert.assertTrue( canonicalPathsForNoXmlFiles.contains( "/filesystemutilities/canonicals/TestFile1.txt" ) );
289 
290         final List<String> canonicalPathsForAllFiles = getRelativeCanonicalPaths( allFiles,
291                 depsPropertiesFile.getParentFile() );
292         Assert.assertTrue( canonicalPathsForAllFiles.contains( "/deps1.properties" ) );
293         Assert.assertTrue( canonicalPathsForAllFiles.contains( "/filefilter/exclusion/TextFile.txt" ) );
294         Assert.assertTrue( canonicalPathsForAllFiles.contains( "/filesystemutilities/TestFile2.txt" ) );
295         Assert.assertTrue( canonicalPathsForAllFiles.contains( "/filesystemutilities/canonicals/TestFile1.txt" ) );
296         Assert.assertTrue( canonicalPathsForAllFiles.contains( "/filefilter/exclusion/AnXmlFile.xml" ) );
297     }

441     @Test
442     public void validateRelativizingPaths() throws Exception
443     {
444 
445         // Assemble
446         final String path = "/project/backend/foobar/my-schema.xsd";
447         final SortedMap<String, String> parentDir2Expected = new TreeMap<>();
448         parentDir2Expected.put( "/", "project/backend/foobar/my-schema.xsd" );
449         parentDir2Expected.put( "", "project/backend/foobar/my-schema.xsd" );
450         parentDir2Expected.put( "/project", "backend/foobar/my-schema.xsd" );
451         parentDir2Expected.put( "/not/a/path", "project/backend/foobar/my-schema.xsd" );
452         parentDir2Expected.put( "/project/", "backend/foobar/my-schema.xsd" );
453 
454         // Act & Assert
455         for ( Map.Entry<String, String> current : parentDir2Expected.entrySet() )
456         {
457 
458             final String expectedMessage = "Given parent dir [" + current.getKey() + "], expected [" + current.getValue() + "] from [" + path + "] but found: ";
459 
460             final String result = FileSystemUtilities.relativize( path, new File( current.getKey() ), true );
461 
462             Assert.assertEquals( expectedMessage + result, current.getValue(), result );
463         }
464     }

466     //
467     // Private helpers
468     //
469 
470     private List<String> getRelativeCanonicalPaths( final List<File> fileList, final File cutoff )
471     {
472 
473         final String cutoffPath = FileSystemUtilities.getCanonicalPath( cutoff ).replace( File.separator, "/" );
474         final List<String> toReturn = new ArrayList<String>();
475 
476         for ( File current : fileList )
477         {
478 
479             final String canPath = FileSystemUtilities.getCanonicalPath( current ).replace( File.separator, "/" );
480             if ( !canPath.startsWith( cutoffPath ) )
481             {
482                 throw new IllegalArgumentException(
483                         "Illegal cutoff provided. Cutoff: [" + cutoffPath + "] must be a parent to CanonicalPath [" + canPath + "]" );
484             }
485 
486             toReturn.add( canPath.substring( cutoffPath.length() ) );
487         }
488         Collections.sort( toReturn );
489 
490         return toReturn;
491     }
