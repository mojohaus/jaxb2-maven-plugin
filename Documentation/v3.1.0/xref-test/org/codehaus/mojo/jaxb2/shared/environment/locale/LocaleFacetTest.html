17  public class LocaleFacetTest {
18  
19      // Shared state
20      private static final Locale FRENCH_LOCALE = Locale.FRENCH;
21      private Locale defaultLocale;
22      private BufferingLog log;
23  
24      @Before
25      public void setupSharedState() {
26          this.log = new BufferingLog(BufferingLog.LogLevel.DEBUG);
27          defaultLocale = Locale.getDefault();
28      }
29  
30      @Test
31      public void validateLocaleParsing() throws MojoExecutionException {
32  
33          // Assemble
34          final String frenchLocaleString = "fr";
35  
36          // Act & Assert
37          final LocaleFacet facet = LocaleFacet.createFor(frenchLocaleString, log);
38          Assert.assertEquals(defaultLocale, Locale.getDefault());
39  
40          facet.setup();
41          Assert.assertEquals(FRENCH_LOCALE.toString(), Locale.getDefault().toString());
42  
43          facet.restore();
44          Assert.assertEquals(defaultLocale.toString(), Locale.getDefault().toString());
45      }
46  
47      @Test
48      public void validateOptimalLocaleFindingIgnoringScripts() throws MojoExecutionException {
49  
50          // Assemble
51          final SortedMap<String, Locale> lang2Locale = new TreeMap<String, Locale>();
52          for (Locale current : Locale.getAvailableLocales()) {
53              lang2Locale.put(current.toLanguageTag(), current);
54          }
55  
56          // Act & Assert
57          for (Map.Entry<String, Locale> current : lang2Locale.entrySet()) {
58  
59              final String language = current.getValue().getLanguage();
60              final String country = current.getValue().getCountry();
61              final String variant = current.getValue().getVariant();
62              final String script = current.getValue().getScript();
63  
64              // Ignore Locales with Scripts.
65              if (script == null) {
66                  Assert.assertSame(Locale.forLanguageTag(current.getValue().toLanguageTag()),
67                          LocaleFacet.findOptimumLocale(language, country, variant));
68              }
69          }
70      }
71
